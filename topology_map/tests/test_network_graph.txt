    # shortest path based on emissions        
    def test_shortest_path_based_on_emissions(self):
        # Assign emissions as node attributes
        emissions_data = {
            'A':40 , 'B': 10, 'C': 60, 'D': 10,  'F': 30, 'G': 10 , 'H': 50, 'I':10       }
        # ALTERNATIVE TEST:  'A':50 , 'B': 10, 'C': 60, 'D': 10,  'F': 30, 'G': 10 , 'H': 90, 'I':30  
        for node, emission in emissions_data.items():
            self.as_graph.graph.nodes[node]['emissions'] = emission

        #weight_func = lambda node: self.as_graph.graph.nodes[node].get('emissions', float('inf'))
        valid_paths = self.as_graph.find_all_valid_shortest_paths('D')

        for dest, path in valid_paths.items():
            total_emission = sum(self.as_graph.graph.nodes[n].get('emissions', 0) for n in path)

        expected_path = ['D', 'B', 'C', 'F', 'I']  # ALTERNATIVE TEST: ['D', 'B', 'C', 'F', 'I']
        self.assertEqual(valid_paths.get('I'), expected_path, f"Incorrect lowest emission path to F")
